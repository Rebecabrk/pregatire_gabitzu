#include <iostream>
#include <fstream>
using namespace std;

ifstream fin("date.txt");

struct ELEV
{
    int cod;
    char nume[51];
    char prenume[51];
    float media;
} e, clasa[30];

int main()
{
    int n;
    fin >> n;
    
    
    for (int i = 1; i <= n; i++)
    {
        fin >> clasa[i].cod;
        fin >> clasa[i].nume;
        fin >> clasa[i].prenume;

    }

    {
        cout << clasa[i].cod << " " << clasa[i].nume << " " << clasa[i].media << endl;
    }

    return 0;
}
